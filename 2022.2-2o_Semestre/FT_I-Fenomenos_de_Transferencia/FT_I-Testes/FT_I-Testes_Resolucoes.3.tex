\begin{questionBox}{ % Q1
} % Q1
    Camisas são equipamentos de troca de calor que se encontram no exterior do reator, sua area de contato é a area externa do reator, possúi maior area de contato que espiras externas e menor que as espiras internas, apesar de ter maior area de contato o fluido é capás de passar por um percurso menor do que uma espira externa que força um percurso por todas as voltas.
    Por ser externa é possivel a troca da camisa sem ter de abrir o reator facilitando manutenção.
\end{questionBox}

\begin{questionBox}1{ % Q1
} % Q1

    \begin{itemize}
        \begin{multicols}{2}
            \item Contra-corrente
            \item \(n_{tubos} = 6\)
            \item \(D_e = 25\,\unit{\milli\metre}\)
            \item \(D_i = 22\,\unit{\milli\metre}\)
        \end{multicols}
        \item Condutividade da parede: \(0.45\,\unit{\watt\,\metre^{-1}\,\kelvin^{-1}}\)
        \item Coeficiente de Filme do ben: \(1.80\,\unit{\kilo\watt\,\metre^{-2}\,\kelvin^{-1}}\)
        
        \item Benzeno
        \begin{itemize}
            \begin{multicols}{2}
                \item \(G_{ben} = 156\,\unit{\kilo\gram\,\minute^{-1}}\)
                \item \(C_{m,ben} = 1.9\,\unit{\kilo\joule\,\kilo\gram^{-1}\,\kelvin^{-1}}\)
                \item \(T_{ben,0} = 330\,\unit{\kelvin}\)
                \item \(T_{ben,1} = 350\,\unit{\kelvin}\)
            \end{multicols}
        \end{itemize}
        
        \item Fluido:
        \begin{itemize}
            \begin{multicols}{2}
                \item \(T_{flu,0} = 280\,\unit{\kelvin}\)

                \item \(\mu_{flu} = 0.0012\,\unit{\kilo\gram\,\metre^{-1}\,\second^{-1}}\)
            \end{multicols}
            \item Condutividade do fluido: \(0.643\,\unit{\watt\,\metre^{-1}\,\kelvin^{-1}}\)
        \end{itemize}
    \end{itemize}

    \begin{BM}
        \left(
            \frac{h_i\,D_i}{K}
        \right)

            \frac{\rho\,D_i\,V}{\mu}

\begin{questionBox}2{ % Q1.1
    Qual o caldal mássico de fluido de arrefecimento necessário sabendo q a temp do fluido n pode ultrapassar 325\,\unit{\kelvin}
} % Q1.1
    \begin{flalign*}
        &
            \min G_{m,flu}
            =\frac{
                Q_{flu}
            }{
                C_{m,flu}
                \,\max{\adif{T}_{flu}}
            }
            = \frac{
                (-Q_{ben})
            }{
                C_{m,flu}
                \,
                (T_{flu,0} - \max{T_{flu}})
            }
            = &\\&
            = \frac{
                (-G_{m,ben}\,\adif{T}_{ben}\,C_{m,ben})
            }{
                C_{m,flu}
                \,
                (T_{flu,0} - \max{T_{flu}})
            }
            % = &\\&
            = -\frac{
                (156/60)
                *(350-330)
                *(1.9*10^3)
            }{
                (4.181*10^3)
                *(280 - 325)
            }
            \cong
            \num{0.525126896808313}
        &
    \end{flalign*}
\end{questionBox}
