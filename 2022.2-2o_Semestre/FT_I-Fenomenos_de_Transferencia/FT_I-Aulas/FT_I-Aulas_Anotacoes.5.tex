\begin{sectionBox}1{Var P at} % S1
    
    \begin{BM}
        L_{eq} = L_{tubo} + \sum_{i\,\in\,\text{acessórios}} L_{i,\,eq}
        \\
        \left(
            -\adif{P}
        \right)_{at}
        = 4\,\Phi\,L_{eq}\,\rho\,v^2/D
        \\
        h_{at}
        = \frac{-\adif{P}_{at}}{\rho\,g}
        = \frac{4\,\Phi\,L_{eq}\,v^2}{D\,g}
    \end{BM}
    
\end{sectionBox}

    \begin{flalign*}
        &

                    \\

                \end{aligned}
            \right)

                \begin{aligned}

                    \\
                    + 2\, & \frac{4\,\Phi\,\rho\,v^2}{D}\,L_{jo}
                \end{aligned}
            \right)
            % = &\\&
            = \frac{4\,\Phi\,\rho\,v^2}{D}
            \,\left(
                \begin{aligned}
                    L_{tb}
                    + 2\,L_{jo}
                \end{aligned}
            \right)
        &
    \end{flalign*}
    
\end{exampleBox}

\begin{sectionBox}1{Var atrito em alargamento súbito} % S2
    
    \begin{BM}
        -\adif{P}_{at}
        = \rho\,\left(v_1-v_2\right)^2/2
        \\
        h_{at}
        = \frac{(v_1-v_2)^2}{2\,g}
    \end{BM}

    \begin{flalign*}
        &
            \adif{G_{m}}
            =\adif{G_{v}\,\rho}
            =\adif{v\,S\,\rho}
            =0
            \implies
            v_1\,S_1 = v_2\,S_2
        &
    \end{flalign*}
    
\end{sectionBox}
