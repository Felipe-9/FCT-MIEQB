\begin{sectionBox}1{First-Order reactions approaching equilibrium} % S1
    
    \begin{center}\huge\color{Emph}\bfseries
        \(\ch{A <=>[\(k\sb{r}\)][\(k'\sb{r}\)] P}\)
    \end{center}
    \begin{BM}
        K
        =\frac{\ch{[A]}\eq}{\ch{[P]}\eq}
        =\frac{k_r}{k'_r}
        \quad
        \begin{cases}
            \ch{[P]}=\ch{[A]}_0-\ch{[A]}
            \\
            \ch{[P]}_0=0
        \end{cases}
        \\
        \ch{[A]}
        = \frac{\ch{[A]}_0}{k'_r+k_r}
        \left(
            k'_r
            +k_r
            \,\exp{\left(
                -(k'_r+k_r)t
            \right)}
        \right)
        \\
        \ch{[A]}\eq
        =\frac{\ch{[A]}_0}{(k_r/k'_r)+1}
        ;\quad
        \ch{[P]}\eq
        =\frac{\ch{[A]}_0}{(k'_r/k_r)+1}
        \\
        v
        =-\odv{\ch{[A]}}{t}
        =k_r\,\ch{[A]}-k'_r\,\ch{[B]}
    \end{BM}

    \begin{center}
        \tikzset{external/remake next=true}

    \begin{sectionBox}*2{integration} % S
        \vspace{-2ex}
        \begin{flalign*}
            &
                \odv{\ch{[A]}}{t}
                = &\\&
                =-k_r\,\ch{[A]}+k'_r\,\ch{[P]}
                \underset{\ch{{P}}_0=0}{=}
                -k_r\,\ch{[A]}+k'_r\,(\ch{[A]}_0-\ch{[A]})
                =-(k'_r+k_r)\,\ch{[A]}
                + k'_r\,\ch{[A]}_0
                \implies &\\[3ex]&
                \implies
                \int_{\ch{[A]}_0}^{\ch{[A]}}{
                    \frac{\odif{\ch{[A]}}}{
                        -(k'_r+k_r)\ch{[A]}
                        + k'_r\ch{[A]}_0
                    }
                }
                = &\\&
                =\int_{
                    -(k'_r+k_r)\ch{[A]}_0
                    +k'_r\ch{[A]}_0
                }^{
                    -(k'_r+k_r)\ch{[A]}
                    +k'_r\ch{[A]}_0
                }{
                    \frac{
                        \frac{\odif{\left(
                            -(k'_r+k_r)\ch{[A]}
                            +k'_r\ch{[A]}_0
                        \right)}}{
                            -(k'_r+k_r)
                        }
                    }{
                        -(k'_r+k_r)\ch{[A]}
                        + k'_r\ch{[A]}_0
                    }
                }
                = &\\&
                =
                \frac{-1}{k'_r+k_r}
                \adif{\left(
                    \ln{
                        -(k'_r+k_r)\ch{[A]}
                        +k'_r\ch{[A]}_0
                    }
                \right)}
                = &\\&
                =
                \frac{-1}{k'_r+k_r}
                \left(
                    \ln{\frac{
                        -(k'_r+k_r)\ch{[A]}
                        +k'_r\ch{[A]}_0
                    }{
                        -(k'_r+k_r)\ch{[A]}_0
                        +k'_r\ch{[A]}_0
                    }}
                \right)
                = &\\&
                =
                \frac{-1}{k'_r+k_r}
                \left(
                    \ln{\frac{
                        -(k'_r+k_r)\frac{\ch{[A]}}{\ch{[A]}_0}
                        +k'_r
                    }{
                        k_r
                    }}
                \right)
                = &\\&
                =
                \frac{-1}{k'_r+k_r}
                \left(
                    \ln{\left(
                        -(k'_r-k_r)\frac{\ch{[A]}}{\ch{[A]}_0}
                        +k'_r
                    \right)}
                    -\ln{k_r}
                \right)
                = &\\[3ex]&
                =\int_{0}^{t}{\odif{t}}
                =t
                \implies &\\[3ex]&
                \implies
                \exp\left(
                    \ln{\left(
                        -(k'_r+k_r)\frac{\ch{[A]}}{\ch{[A]}_0}
                        +k'_r
                    \right)}
                \right)
                =
                -(k'_r+k_r)\frac{\ch{[A]}}{\ch{[A]}_0}
                +k'_r
                = &\\[1.5ex]&
                = \exp{\left(
                    -(k'_r+k_r)t+\ln{k_r}
                \right)}
                = k_r\exp{\left(
                    -(k'_r+k_r)t
                \right)}
                \implies &\\[3ex]&
                \implies
                \ch{[A]}
                =
                \frac{\ch{[A]}_0}{k'_r+k_r}
                \left(
                    k'_r
                    +k_r
                    \,\exp{\left(
                        -(k'_r+k_r)t
                    \right)}
                \right)
                &\\[3ex]&
                \lim_{t\to\infty}\ch{[A]}
                = \frac{\ch{[A]}_0}{k'_r+k_r}
                \left(
                    k'_r
                    +k_r
                    *0
                \right)
                = \frac{\ch{[A]}_0\,k'_r}{k'_r+k_r}
            &
        \end{flalign*}
    \end{sectionBox}

            \begin{BM}
                K=\prod_{i=1}^{n}{\frac{k_i}{k'_i}}
            \end{BM}
        \end{center}
        
    \end{sectionBox}
    
\end{sectionBox}
