\begin{sectionBox}1{Zeroth-Order reaction \ch{A -> P}}

    \begin{BM}
        v
        = \odv{\ch{[P]}}{t}
        =-\odv{\ch{[A]}}{t}
        = k_r
        \\
        \ch{[A]}
        =\begin{cases}
            \ch{[A]}_0-k_r\,t
            \quad& t\leq\ch{[A]}_0/k_r
            \\
            0\quad&t>\ch{[A]}_0/k_r
        \end{cases}
    \end{BM}
    \begin{sectionBox}*3{Intregration} % S
        
        \begin{flalign*}
            &
                \odv{\ch{[A]}}{t}=-k_r
                \implies &\\&
                \implies
                \int{\odif{\ch{[A]}}}
                =\adif{\ch{[A]}}
                =\ch{[A]}-\ch{[A]}_0
                % = &\\&
                =-\int{k_r\odif{t}}
                =-k_r\adif{t}
                =-k_r\,t
                \implies &\\&
                \implies
                \ch{[A]}=\ch{[A]}_0-k_r\,t
            &
        \end{flalign*}
        
    \end{sectionBox}

\begin{sectionBox}1{First-order reaction \ch{A -> P}} % S
    
    \begin{BM}
        v
        = \odv{\ch{[P]}}{t}
        =-\odv{\ch{[A]}}{t}
        = k_r\,\ch{[A]}
        \\
        \ln{\frac{\ch{[A]}}{\ch{[A]}_0}}
        = -k_r\,t
        \iff
        \ch{[A]}=\ch{[A]}_0\,e^{-k_r\,t}
    \end{BM}
    \begin{sectionBox}*3{Intregration} % S
        \begin{flalign*}
            &
                \odv{\ch{[A]}}{t}
                =-k_r\,\ch{[A]}
                \implies &\\&
                \implies
                \int{\frac{\odif{\ch{[A]}}}{\ch{[A]}}}
                = \adif{\ln{\ch{[A]}}}
                = \ln{\frac{\ch{[A]}}{\ch{[A]}_0}}
                % = &\\&
                = -\int{k_r\,\odif{t}}
                = -k_r\,t
                \implies &\\&
                \implies
                \ch{[A]}=\ch{[A]}_0\,e^{-k_r\,t}
            &
        \end{flalign*}
    \end{sectionBox}

\begin{sectionBox}1{Second-order reaction \ch{2 A -> P}} % S

    \begin{BM}
        v
        = \odv{\ch{[P]}}{t}
        =-\odv{\ch{[A]}}{t}
        = k_r\,\ch{[A]}^2
        \\
        \frac{1}{\ch{[A]}}-\frac{1}{\ch{[A]_0}}=k_r\,t
        \iff
        \ch{[A]} = \frac{\ch{[A]}_0}{1+k_r\,t\,\ch{[A]}_0}
    \end{BM}
    \begin{sectionBox}*2{Intregration} % S
        \begin{flalign*}
            &
                \odv{\ch{[A]}}{t}
                =-k_r\,\ch{[A]}^2
                \implies &\\&
                \implies
                \int{\frac{\odif{\ch{[A]}}}{\ch{[A]}^2}}
                =\adif{\left(-\ch{[A]}^{-1}\right)}
                =\ch{[A]}_{0}^{-1}
                -\ch{[A]}^{-1}
                = &\\&
                =-\int{k_r\,\odif{t}}
                =-k_r\,t
                \implies &\\&
                \implies
                \ch{[A]}^{-1}
                =k_r\,t+\ch{[A]}_{0}^{-1}
                \iff
                \ch{[A]}=\frac{\ch{[A]}_0}{1+k_r\,t\,\ch{[A]}_0}
            &
        \end{flalign*}
    \end{sectionBox}

\begin{sectionBox}1{Second-order reaction \ch{A + B -> P}}
    
    \begin{BM}
        v
        = \odv{\ch{[P]}}{t}
        =-\odv{\ch{[A]}}{t}
        =-\odv{\ch{[B]}}{t}
        = k_r\,\ch{[A][B]}
        \\
        \ln{
            \frac
            {\ch{[B]/[B]_0}}
            {\ch{[A]/[A]_0}}
        }
        =(\ch{[B]}_0-\ch{[A]}_0)\,k_r\,t
        \quad :\ch{{A}}_0\neq\ch{[B]}_0
    \end{BM}

    \paragraph*{Note:} Behaves as \ch{2 A -> P} if \(\ch{[A]}_0=\ch{[B]}_0\)

    \begin{sectionBox}*3{Integration} % S
        \begin{center}
            \vspace{1ex}
            \begin{tabular}{*{2}{C} | C}
                \toprule
                
                    \ch{[A]}
                    & \ch{[B]}
                    & \ch{[P]}
                
                \\\midrule
                
                      \ch{[A]}_0
                    & \ch{[B]}_0
                    & 0
                
                    \\\ch{[A]}_0-x
                    & \ch{[B]}_0-x
                    & x
                
                \\\bottomrule
            \end{tabular}
            \vspace{-2ex}
        \end{center}
        \begin{flalign*}
            &
                \odv{\ch{[A]}}{t}
                =\odv{\ch{[A]}_0-x}{t}
                =-\odv{x}{t}
                = &\\&
                =-k_r\,\ch{[A][B]}
                =-k_r\,(\ch{[A]}_0-x)(\ch{[B]}_0-x)
                \implies &\\&
                \implies
                \int_{0}^{x}{
                    \frac{\odif{x}}{
                        (\ch{[A]}_0-x)(\ch{[B]}_0-x)
                    }
                }
                = \frac{1}{\ch{[B]}_0-\ch{[A]}_0}
                \left(
                    \ln{\frac{\ch{[A]}_0}{\ch{[A]}_0-x}}
                    -\ln{\frac{\ch{[B]}_0}{\ch{[B]}_0-x}}
                \right)
                = &\\&
                \underset{\ch{[A]}_0\neq\ch{[B]}_0}{=}
                \frac{1}{\ch{[B]}_0-\ch{[A]}_0}
                \left(
                    \ln{\frac{\ch{[A]}_0}{\ch{[A]}}}
                    -\ln{\frac{\ch{[B]}_0}{\ch{[B]}}}
                \right)
                = \frac{1}{\ch{[B]}_0-\ch{[A]}_0}
                \ln{\frac{
                    \ch{[B]}/\ch{[B]}_0
                }{
                    \ch{[A]}/\ch{[A]}_0
                }}
                = &\\&
                = \int_{0}^{t}{k_r\,\odif{t}}
                = k_r\,t
                \implies &\\&
                \implies
                \ln{\frac{
                    \ch{[B]}/\ch{[B]}_0
                }{
                    \ch{[A]}/\ch{[A]}_0
                }}
                = (\ch{[B]}_0-\ch{[A]}_0)\,k_r\,t
            &
        \end{flalign*}
    \end{sectionBox}
    
\end{sectionBox}

\begin{sectionBox}1{n-th order reaction \ch{n A -> P}} % S
    \begin{BM}
        v
        = \odv{\ch{[P]}}{t}
        =-\odv{\ch{[A]}}{t}
        = k_r\,\ch{[A]}^n
        \\
        k_r\,t
        =\frac{1}{n-1}
        \left(
            (\ch{[A]}_0-\ch{[P]})^{1-n}
            -\ch{[A]}_0^{1-n}
        \right)
        ;\quad n\geq2
    \end{BM}
    \begin{sectionBox}2{Half life} % S
        \begin{BM}
            t_{1/2}
            = \frac{2^{n-1}-1}{(n-1)\,k_r\,\ch{[A]}_0^{n-1}}
            \quad:n>1
        \end{BM}
    \end{sectionBox}
\end{sectionBox}
