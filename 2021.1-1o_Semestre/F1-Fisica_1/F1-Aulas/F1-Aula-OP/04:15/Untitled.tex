\subsection{Trabalho pela mola}
\begin{flalign*}
&
	w_{F_k} = \int \vec F_{k}\cdot\vec d
	=
		-k\,\int d\delta d
	=
		-k\,d^2/2
&
\end{flalign*}

\subsection{Velocidade do bloco no momento que encosta na mola}
\begin{flalign*}
&
	m\,v^2/2 = k\,d^2/2
	\implies
		v = \sqrt{k\,d^2/m}
&
\end{flalign*}

\subsection{Altura que partil}
\begin{flalign*}
&
	m\,g\,h = m\,v^2/2
	\implies
		h = v^2/(2\,g)
&
\end{flalign*}
