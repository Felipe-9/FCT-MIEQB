\end{flalign*}

\begin{flalign*}

;\	a = v^2/R
\implies &\\& \implies
	\vv{F_{N}}
=	(m\,(\sqrt{3\,g\,R})^2/R-m\,g)\,-\hat\jmath
=	&\\&
=	-2\,m\,g\,\hat\jmath
=	-2*5.00*\num{9.80665}
\cong
	\qty{-0.0980665}{\newton}\,\hat\jmath
&
\end{flalign*}
