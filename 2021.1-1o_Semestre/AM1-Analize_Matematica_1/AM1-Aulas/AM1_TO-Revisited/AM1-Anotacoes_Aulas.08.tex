\begin{flalign*}
&
	u_{(n)}\to l_1
\land
	v_{(n)}\to l_2
\implies	&\\&
\implies
	0
\leq	|u_{(n)}\,v_{(n)}-l_1\,l_2|
=	&\\&
=	|(u_{(n)}-l_1)\,v_{(n)}+v_{(n)}\,l_1-l_1\,l_2|
=	&\\&
=	|(u_{(n)}-l_1)\,v_{(n)} + l_1\,(v_{(n)}-l_2)|
\leq	&\\&
\leq	|(u_{(n)}-l_1)|\,|v_{(n)}| + |l_1|\,|(v_{(n)}-l_2)|
\to 0
\implies	&\\&
\implies
	u_{(n)}*v_{(n)}\to l_1*l_2
&
\end{flalign*}

\end{flalign*}

\section{Exercicios}

\begin{questionBox}3{$ u_{(n)}=\sqrt{n^2+2\,n}-n $}
\begin{flalign*}
&
=	\frac{n^2+2\,n-n^2}{\sqrt{n^2+2\,n}+n}
=	\frac{2}{\sqrt{1+2/n}+1}
\to	&\\&
\to	\frac{2}{\sqrt{1}+1}
=	1
&
\end{flalign*}
\end{questionBox}

\begin{questionBox}3{$ a_{(n)} = \frac{n+\cos(n)}{n\,\ln(n+1)} $}
\begin{flalign*}
&
	\lim_{n\to\infty} a_{(n)}
=	\lim_{n\to\infty} \frac{1}{\ln(n+1)}
+	\lim_{n\to\infty} \frac{\cos(n)}{n}
*	&\\&
*	\lim_{n\to\infty} \frac{1}{\ln(n+1)}
=	0 + 0*0 = 0
&
\end{flalign*}
\end{questionBox}

\begin{questionBox}3{$ 
	b_{(n)} 
= 	\left( 
	\frac{n^2-2\,n+1}{n^2+2\,n+1} 
	\right)^n 
$}
\begin{flalign*}
&
	\lim_{n\to\infty} b_{(n)}
=	\lim_{n\to\infty} 
	\left( 
		\frac{(n-1)^2}{(n+1)^2} 
	\right)^n
=	\lim_{n\to\infty}
	\left( 
		\frac{n-1}{n+1} 
	\right)^{2\,n}
=	&\\&
=	\left(
		\frac{\lim_{n\to\infty}(1-1/n)^n}
			{\lim_{n\to\infty}(1+1/n)^n} 
	\right)^2
=	\left(
		\frac{e^{-1}}
			{e^1} 
	\right)^2
=	e^{-4}
&
\end{flalign*}
\end{questionBox}
