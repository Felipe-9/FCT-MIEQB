\begin{sectionBox}1{Exercicio: $f(x)=x(x^2+1)^3$}

\begin{flalign*}
&
	F(x)=\int x(x^2+1)^3\,\mathrm{d}x
=	0.5\,\int u'(u)^3\,\mathrm{d}x\,
=	&\\&
=	u^4/4+c
\impliedby
	F'(x)= 3\,u^3\,u'=f(x)
&
\end{flalign*}

\end{sectionBox}

\begin{sectionBox}1{Primitivas por substituição}
\begin{BM}
	F'(x(t)) = f(x(t))\,\odv{x}{t}
\end{BM}
\end{sectionBox}

\begin{sectionBox}2{Exemplo: $f(x)=e^x/(1+e^{2\,x})$}
\begin{flalign*}
&
	F'(x)
=	\int\frac{e^x}{1+e^{2\,x}}\,\mathrm{d}x
=	\int\frac{t}{1+t^2}\odv{\ln(t)}{t}\,\mathrm{d}t\,
=	&\\&
=	\int(1+t^2)^{-1}\,\mathrm{d}t
=	\arctan(t)+c
=	\arctan(e^x)+c
&
\end{flalign*}
\end{sectionBox}

\begin{sectionBox}2{Exemplo: $f(x)=e^{\sqrt{x}}/\sqrt{x}$}
\begin{flalign*}
&
	F(x)
=	\int e^{\sqrt{x}}\,\sqrt{x}^{-1}\,\mathrm{d}x
=	\int e^{t}\,t^{-1}\,\odv{t^2}{t}\,\mathrm{d}t\,
=	&\\&
=	\int 2\,e^{t}\,\mathrm{d}t
=	2\,e^{t}+c=2\,e^{\sqrt{x}}+c
&
\end{flalign*}
\end{sectionBox}

\begin{sectionBox}2{Exemplo: $f(x)=\sqrt{x-1}/x$}
\begin{flalign*}
&
	F(x)
=	\int\sqrt{x-1}\,x^{-1}\,\mathrm{d}x
=	\int \frac{t}{t^2+1}\,\odv{(t^2+1)}{t}\,\mathrm{d}t\,
=	&\\&
=	\int \frac{2\,(t^2+1-1)}{t^2+1}\,\mathrm{d}t
=	\int 2\,\mathrm{d}t
-	\int \frac{2}{t^2+1}\,\mathrm{d}t\,
=	&\\&
=	2\,t-2\arctan(t)+c\,
=	&\\&
=	2(\sqrt{x-1}-\arctan(\sqrt{x-1}))+c
&
\end{flalign*}
\end{sectionBox}
