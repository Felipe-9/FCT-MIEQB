                    \leq 2^{m+1-1} = 2^{m}

                    \implies
                    2
                    \leq 2^{m+1}-2^m
                    = 2^m(2-1)
                    = 2^m
                &
            \end{flalign*}
        \end{exampleBox}
    \end{exampleBox}
