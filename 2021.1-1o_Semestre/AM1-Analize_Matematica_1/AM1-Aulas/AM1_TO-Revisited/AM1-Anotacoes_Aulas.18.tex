\part{27/04 - Derivadas}

\section{Definição}
\begin{BM}
	f:\mathbf{I} \to \mathbb{R}
\land
	a\in\text{\hyperref[interior]{int}}(\mathbf{I})
\end{BM}\relax

\begin{BM}
	D\in\mathbb{R}:
	D = f'(a) = \odv{f(a)}{x}
=	\lim_{x\to a}\frac{f(x)-f(a)}{x-a}
=	\lim_{h\to 0}\frac{f(a+h)-f(a)}{h}
\end{BM}
\label{derivadas}

\section{Definição Alternativa}

\begin{BM}
	\exists\,f'(a)
\impliedby
	\exists\,\mathrm{D} \in\mathbb{R},
\\
\quad Z:\vizinhanca{0}\to\mathbb{R}:
\\
	f(x) = f(a) + D\,(x-a)
+	\\
+	(x-a)\,Z(x-a)
\end{BM}

\section{Definição derivada para funções de imagens abertas}
