\begin{sectionBox}1{Introdução: Cálculo de Areas}

\begin{BM}
	\text{Area}_{\min}(f(x))
=	\underline{\mathrm{S}}(f(x),p)
=	\sum_{k=0}^{n}\inf(f([x_k,\,x_k+1]))
\end{BM}\relax

\begin{BM}
	\text{Area}_{\max}(f(x))
=	\overline{\mathrm{S}}(f(x),p)
=	\sum_{k=0}^{n}\sup(f([x_k,\,x_k+1]))
\end{BM}

\begin{BM}
	\underline{\int_{a}^{b}}f(x)\,\mathrm{d}x
=	\sup\left(
		\underline{\mathrm{S}}(f,p):
		p \text{ partição de } [a,b]
	\right)
\end{BM}\relax

\begin{BM}
	\overline{\int_{a}^{b}}f(x)\,\mathrm{d}x
=	\inf\left(
		\overline{\mathrm{S}}(f,p):
		p \text{ partição de } [a,b]
	\right)
\end{BM}\relax

\begin{BM}
	\overline{\int_{a}^{b}}f(x)\,\mathrm{d}x
\geq\underline{\int_{a}^{b}}f(x)\,\mathrm{d}x
\end{BM}\relax

\label{Rieman}
\begin{sectionBox}1{Definição: Integração de Rieman}
\begin{BM}
	f
	\begin{cases}
		f:[a,b]\to\mathbb{R}\,\land
	\\	\text{limitada}
	\end{cases}
\end{BM}\relax

\begin{flalign*}
&
	f \text{ é integrável em Rieman}
\impliedby &\\&
\impliedby
	\underline{\int_{a}^{b}}f(x)\,\mathrm{d}x
=	\overline{\int_{a}^{b}}f(x)\,\mathrm{d}x
&
\end{flalign*}
\end{sectionBox}

\begin{sectionBox}1{Critério necessário e suficiente
para a integralidade de uma função}
\begin{BM}
	f
	\begin{cases}
		f:[a,b]\to\mathbb{R}\,\land
	\\	\text{Limitada}
	\end{cases}
\end{BM}

\begin{flalign*}
&
	\exists\,P_n\subset[a,b]:
	\lim_{n\to\infty}
	\left(
		\overline{\mathrm{S}}(f,P_n)
	-	\underline{\mathrm{S}}(f,P_n)
	\right)
=	0
&
\end{flalign*}
\end{sectionBox}
