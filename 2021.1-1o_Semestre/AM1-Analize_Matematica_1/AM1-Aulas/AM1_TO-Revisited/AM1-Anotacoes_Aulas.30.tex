\begin{sectionBox}3{$\overline{ \mathrm{S}}(f,P_n)$}
\begin{flalign*}
&
=	\sum_{k=0}^{n-1}\sup(f([k,k+1]))\frac{b-a}{n}
=	&\\&
=	\sum_{k=0}^{n-1}f(k+1)\frac{b-a}{n}
&
\end{flalign*}
\end{sectionBox}

\begin{sectionBox}3{$\underline{ \mathrm{S}}(f,P_n)$}
\begin{flalign*}
&
=	\sum_{k=0}^{n-1}\inf(f([k,k+1]))\frac{b-a}{n}
=	\sum_{k=0}^{n-1}f(k)\frac{b-a}{n}
&
\end{flalign*}
\end{sectionBox}

\begin{flalign*}
&
	\overline{ \mathrm{S}}(f,P_n)
-	\underline{\mathrm{S}}(f,P_n)
=	\frac{b-a}{n}
	\sum_{k=1}^{n}
	(f(k+1)-f(k))
=	&\\&
=	\frac{b-a}{n}
	(f(b)-f(a))
&
\end{flalign*}

\begin{sectionBox}3{Limite}
\begin{flalign*}
&
	\lim_{n\to\infty}
	\frac{b-a}{n}
	(f(b)-f(a))
=	0
&\\&
	f(x) \text{ é integrável por Rieman}
&
\end{flalign*}
\end{sectionBox}

\end{sectionBox}\relax

\begin{sectionBox}1[0mm]{Contra Exemplo}
\begin{BM}
	f
	\begin{cases}
	f:[0,1]\to\mathbb{R}\,\land
	\\	f(x)
	=	\begin{cases}
			1&\quad x\in\mathbb{Q}
		\\	0&\quad x\not\in\mathbb{Q}
		\end{cases}
	\end{cases}
\end{BM}

\begin{flalign*}
&
	\sup(f([k,k+1]))=1
	\land
	\inf(f([k,k+1]))=0
&\\&
	\forall\,k\in P_n\text{ Partição de } [a,b]
	\therefore
	\overline{ \mathrm{S}}(f,P_n)
-	\underline{\mathrm{S}}(f,P_n)
	\neq 0
&
\end{flalign*}
\end{sectionBox}

\begin{sectionBox}1{Diâmetro de uma partição}
\begin{BM}
	\delta(P)=\max(|x_{k+1}-x_k|)
	\\
	\quad\forall\,k\in[0,n-1]
\end{BM}
\end{sectionBox}

\begin{sectionBox}1{Soma de Riemam}
\begin{BM}[flalign*]
&
	\int_{a}^{b} f(x)\,\mathrm{d}x
=	&\\&
=	\lim_{\delta(P)\to0}
	\sum_{k=0}^{n-1}
		f(x'_k)(f(x_{k+1})-f(x_k))
	&\\& x'_k\in[x_k,x_{k+1}]
&
\end{BM}
\end{sectionBox}

\section*{Propriedades da soma de Rieman}

\begin{sectionBox}3[0mm]{}
\begin{BM}
	\int_{a}^{b}(k\,f(x)+g(x))\,\mathrm{d}x
=	\\
=	k\int_{a}^{b}f(x)\,\mathrm{d}x
+	\int_{a}^{b}g(x)\,\mathrm{d}x
\end{BM}
\end{sectionBox}

\begin{sectionBox}3[0mm]{}
\begin{BM}
	\int_{a}^{b}f(x)\,\mathrm{d}x
	\leq
	\int_{a}^{b}g(x)\,\mathrm{d}x
	\\
	\quad\forall\,\{f,g\}:f(x)\leq g(x)\quad\forall\,x\in[a,b]
\end{BM}
\end{sectionBox}

\begin{sectionBox}3[0mm]{}
\begin{BM}
	\left|
		\int_{a}^{b}
		f(x)\,\mathrm{d}x
	\right|
	\leq
	\int_{a}^{b}
	\left|
		f(x)\,\mathrm{d}x
	\right|
\end{BM}
\end{sectionBox}
