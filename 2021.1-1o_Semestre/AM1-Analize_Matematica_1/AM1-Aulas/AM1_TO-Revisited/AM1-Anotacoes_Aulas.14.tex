\section{Limite notável: $(e^x-1)/x$}

\BMold{\lim_{x\to0} \frac{e^x-1}{x}=1}\relax

\begin{flalign*}
&
=	\lim_{x\to0} \frac
		{\left(1 + \sum_{n=1}^{\infty} \frac{x^n}{n!}\right) - 1}
		{x}
=	&\\&
=	\lim_{x\to0} \sum_{n=1}^{\infty} \frac{x^{(n-1)}}{n!}
=	\lim_{x\to0} 1 + \sum_{n=2}^{\infty} \frac{x^{(n-1)}}{n!}
=	1
&
\end{flalign*}

\end{minipage}
