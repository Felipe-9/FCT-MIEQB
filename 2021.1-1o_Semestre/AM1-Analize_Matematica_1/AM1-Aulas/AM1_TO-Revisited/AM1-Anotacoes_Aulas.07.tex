\section{Lema $
	\lim\limits_{n\to\infty}u_{(n)}
\leq	\lim\limits_{n\to\infty}v_{(n)}
$}
\begin{flalign*}
&
	u_{(n)} \text{ e } v_{(n)} \text{ são convergentes}\,
\land	&\\&
\land
	u_{(n)} \leq v_{(n)}\quad\forall\,n\in\mathbb{N}
\implies	&\\&
\implies
	\lim\limits_{n\to\infty}u_{(n)}
\leq	\lim\limits_{n\to\infty}v_{(n)}
&
\end{flalign*}

\section{Lema das sucessões enquadradas}
\label{lema das sucessoes enquadradas}

\begin{flalign*}
&
	\{u_{(n)}, v_{(n)}, w_{(n)}\}:\mathbb{N}\to\mathbb{R}
\,\land	&\\&
\land
	v_{(n)}\leq u_{(n)}\leq w_{(n)}
	\quad\forall\,n\in\mathbb{N}
\,\land	&\\&
\land
	\exists\,l\in\mathbb{R}
	:\{v_{(n)},w_{(n)}\} \converge l
\implies	&\\&
\implies
	u_{(n)} \converge l
&
\end{flalign*}


\subsection{Exemplo: $ u_{(n)}=\sin(n)/n $}

\begin{flalign*}
&
	\frac{-1}{n}
\leq	\frac{\sin(n)}{n}
\leq	\frac{1}{n}
;\
	\left\{\frac{-1}{n}, \frac{1}{n}\right\}\to 0
\implies	&\\&
\implies
	\frac{\sin(n)}{n}\to0
&
\end{flalign*}



\section{Lema $ u_{(n)}*v_{(n)}\to 0 $}

\begin{flalign*}
&
	u_{(n)} \text{ é limitada}
\land
	v_{(n)} \to 0
\implies	&\\&
\implies
	0
\leq	|u_{(n)}*v_{(n)}|=|u_{(n)}|*|v_{(n)}|
\leq	&\\&
\leq	\hyperref[majorante]{\text{Majorante}(u_{n})}
	*0=0
\implies	&\\&
\implies
	u_{(n)}*v_{(n)}\to 0
&
\end{flalign*}

\subsection{Exemplo: $S_{(n)} = \sum_{k=1}^n\frac{\sqrt{k}}{n^2}$}

\begin{flalign*}
&
	0
\leq	S_{(n)}
=	\sum_{k=1}^n\frac{\sqrt{k}}{n^2}
\leq	\frac{n\,\sqrt{n}}{n^2}
=	\frac{1}{\sqrt{n}}
\to 0
\implies	&\\&
\implies
	S_{(n)}\to 0
&
\end{flalign*}


\subsection{Exercicio: $\frac{2^n-n}{3^n+1}$}

\begin{flalign*}
&
	0
\leq	\frac{2^n-n}{3^n+1}
\leq \frac{2^n}{3^n}
=	(2/3)^n
\to 0
\implies	&\\&
\implies
	\frac{2^n-n}{3^n+1}\to0
&
\end{flalign*}

\end{flalign*}
