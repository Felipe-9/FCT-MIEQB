	\lim_{n\to\infty}\frac{n^2+n\sqrt{n}}{3n^2+3}
$}

&

		\lim_{n\to\infty}f_{a(n)} = 1/3
&
\end{flalign*}

% Q3 - c)
\subsection{$
	\lim_{n\to\infty}\frac{n\,\sqrt[3]{8n+1}}{(n^{2/3}+1)^2}
$}
\begin{flalign*}
&
	\frac{n\,\sqrt[3]{8n+1}}
		  {(n^{2/3}+1)^2    } 
	= \frac{\sqrt[3]{8n^4+n^3}}
			 {(n^{2/3}+1)^2     } 
	* \frac{n^{4/3}} 
			 {n^{4/3}} 
	= \cfrac{
		\sqrt[3]{ 
			\cfrac{8n^4}
			 	  {n^4 }
		 + \cfrac{n^3}
			 	  {n^4}
		}
	}
	{
		\left( 
			\cfrac{n^{2/3}}
				  {n^{2/3}} 
			+ \cfrac{1      }
					 {n^{2/3}} 
		\right)^2
	} 
	= \frac{ \sqrt[3]{ 8+n^{-1} }      }
			 { \left( 1+n^{-2/3} \right) }
	= f_{c(n)}
	\implies &\\& \implies
		\lim_{n\to\infty} f_{c(n)} 
	= \frac{\sqrt[3]{8}}
			 {1          }
	= 2
&
\end{flalign*}
