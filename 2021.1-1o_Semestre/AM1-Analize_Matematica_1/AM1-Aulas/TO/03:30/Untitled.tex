% E1
\subsection{E1}
\bm{ u_n=\sqrt{n^2+2n}-n }
\begin{flalign*}
&	u_n= n(\sqrt{1+2/n}-1)
		= \frac{n(1+2/n-1)}{\sqrt{1+2/n}+1}
		= \frac{2}{\sqrt{1+2/n}+1}
		\to 1 &
\end{flalign*}

% E2
\subsection{E2}
\bm{ a_n=\frac{n+\cos(n)}{n\,\ln(n+1)};\ 
	  b_n=\left(\frac{n^2-2\,n+1}{n^2+2n+1}\right)^n }
\begin{flalign*}
&	a_n = \frac{n+\cos(n)}{n\,\ln(n+1)}
		 = \frac{1+\cos(n)/n}{\ln(n+1)}
		 \to \infty &\\
&	b_n = \left(\frac{n^2-2\,n+1}{n^2+2n+1}\right)^n
		 = \left( \frac{(n+1)^2}{(n-1)^2} \right)
		 = \left( \frac{n+1}{n-1}  \right)^{2\,n}
		 = \left( \frac{n-1+2}{n-1}\right)^{2\,n} = &\\
&		 = \left( 1+\frac{2}{n-1}  \right)^{2\,n} 
		 = \left(\left( 1+\frac{2}{n-1} \right)^{n-1}\right)^2
		 	\,\left( 1+\frac{2}{n-1} \right)^2
		 \to e^4 &
\end{flalign*}
