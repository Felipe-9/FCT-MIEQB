% H2 1o
\begin{flalign*}	
&
	\frac{40\,\mol[1o]{H2}}{100\,\mol[1i]{H2}}
	\,70\%\,\mol[1i]{H2}
\cong
	28\%\,\mol[1o]{H2}
&	
\end{flalign*}

% CO2 1o
\begin{flalign*}
&
	28\%\,\mol[1i]{CO2}
	-
	\frac{ 1\,\mol[1i]{CO2} }{ 3\,\mol[1i]{H2} }
	\,\frac{ 60\,\mol[1i]{H2} }{ 100\,\mol[1i]{H2} }
	\,70\%\,\mol[1i]{H2}
\cong
	14\%\,\mol[1o]{CO2}
&
\end{flalign*}

% CH3OH 1o
\begin{flalign*}
&
	\frac{ 60\,\mol[1o]{CH3OH} }{ 100\,\mol[1o]{CH3OH} }
	\,\frac{ 1\,\mol[1o]{CH3OH} }{ 3\,\mol[1i]{H2} }
	\,70\,\mol[1i]{H2}
\cong
	14\%\,\mol[1o]{CH3OH}
&
\end{flalign*}

% H2O 1o
\begin{flalign*}
&
	\frac{1\,\mol[1o]{H2O}}{1\,\mol[1o]{CH3OH}}
	\,14\,\mol[1o]{CH3OH}
\cong
	14\,\mol[1o]{H2O}
&
\end{flalign*}
