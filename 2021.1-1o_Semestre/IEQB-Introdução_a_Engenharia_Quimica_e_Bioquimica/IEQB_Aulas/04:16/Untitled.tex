\subsection{Porcentagem em excesso}
\begin{align*}
&
	\%\text{Excesso} = 100\,(N/N_s - 1)
	&\\&
	N = \text{Quantidade de mols do reagente}
	&\\&
	N_s = \text{Quantidade necessária de mols de reagente}
&
\end{align*}

\subsection{Exemplo }

\center{

% mol C3H3N
\begin{flalign*}
& 
	\frac{ 1\,\mol{out\, C3H3N} }
		{ 1\,\mol{C2H6}}
	*
	\frac{30	\,\mol{C2H6}}
		{100	\,\mol{C2H6}}
	*
	\frac{ 0.1*100\,\mol{C2H6} }
		{h}
	=
		3.0\, \mol{out\,C3H3N}/h
&
\end{flalign*}

% mol H2O
\begin{flalign*}
& 
	\frac{ 3\,\mol{H2O}		}
		{ 1\,\mol{C3H3N}	}
	*
	\frac{ 3\,\mol{C3H3N}	}
		{ h				}
	=
		9.0\,\mol{H2O}/h
&
\end{flalign*}

% mol NH3
\begin{flalign*}
& 
	0.12*100\,\mol{NH3}/h
	-
	\frac{ 1\,\mol{NH3}		}
		{ 1\,\mol{C2H6}	}
	*
	\frac{ 30 \,\mol{C2H6}	}
		{ 100\,\mol{C2H6}	}
	*
	\frac{ 0.10*100\,\mol{C2H4} }
		{ h }
	=
		9.0\,\mol{NH3}/h
&
\end{flalign*}

% mol O2
\begin{flalign*}
&
	0.78*100\,\mol{O2}/h
	-
	\frac{ 3\,\mol{O2}		}
		{ 2\,\mol{C2H4}	}
	*
	\frac{ 30 \,\mol{C2H4}	}
		{ 100\,\mol{C2H4}	}
	*
	\frac{ 0.10*100\,\mol{C2H4} }
		{ h }
	\cong
		74\,\mol{O2}/h
&
\end{flalign*}

		7.0\,\mol{C2H6}/h
&
\end{flalign*}
