\subsection{}
\bm{
	x_{i1} &= 0.10 \text{ etanol}; \\
	y_{i1} &= 0.90 \text{ água}; \\
	m_{i1} &= 1000\,kg/h; \\
	x_{o1} &= 0.60 \text{ etanol}; \\
	m_{o1} &= 0.10 * m_{i1}\,kg/h
}
\begin{flalign*}
&
	m_{o2} = m_{i1} - m_{o1}
	= 1000 - 0.10 * 1000
	= 900 Kg/h
&\\&
	x_{o2}\,m_{o2} = x_{i1}\,m_{i1} - x_{o1}\,m_{o1}
	\implies
		x_{o2} 
	= 
		\frac{0.10*1000-100*0.60}{900}
	\cong	
		0.0444\,p/p
&\\&
	y_{o2}\,m_{o2} 
	= y_{i1}\,m_{i1}-y_{o1}\,m_{o1}
	\implies
		y_{o2} 
	= 
		\frac{0.90*1000-0.40*100}{900}
	\cong
		0.956\,p/p
&
\end{flalign*}
