    Sequencia de \emph{nucleótidos}, exemplo: \emph{DNA}
    \paragraph*{\color{GraphA15}Polissacarídeos:}
    \emph{Monossacarídeos} ligados por por \emph{ligações glicosídicas}

\begin{sectionBox}2{\color{Graph32}Elastomeros} % S

\end{sectionBox}
