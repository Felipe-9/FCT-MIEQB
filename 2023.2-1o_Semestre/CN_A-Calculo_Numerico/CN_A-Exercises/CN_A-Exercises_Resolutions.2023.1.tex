\begin{questionBox}1{} % Q1

  Indique, justificando, o número de casas decimais significativas e o número de algarismos significativos que se pode garantir para cada uma das aproximações seguintes.
  \begin{enumerate}[label=\alph{enumi})]
    \begin{multicols}{3}
      \item 108.1
      \item 94.23 (8)
      \item 61.124 (0.5)
      \item 5.03 (20)
      \item 206.1\pm0.02
    \end{multicols}
  \end{enumerate}

  \answer{}

  \setcounter{subquestion}{2}
  \begin{questionBox}2b{ 61.124(0.5) } % Q1.3
    \begin{gather*}
      \begin{cases}
        \hat{X}=61.124
        \\
        \myvert{\eta_X}=0.5\E-3
        \\
        \text{Casas decimais significativas}
        = 3
      \end{cases}
      \end{gather*}
  \end{questionBox}

  \setcounter{subquestion}{4}
  \begin{questionBox}2b{ 206.1\pm 0.02 } % Q1.5
    \begin{gather*}
      \begin{cases}
        \hat{X}=206.1
        \\
        \myvert{\eta_X}=0.02\leq 0.5\E-2
        \\
        \text{Casas decimais significativas}
        =2
      \end{cases}
      \end{gather*}
  \end{questionBox}

  \begin{questionBox}2b{ 717.0 } % Q
    \begin{gather*}
      \begin{cases}
        \hat{X}=717.0
        \\
        \myvert{\eta_X}=0.5\E-1
        \\
        \text{Casas decimais significativas}
        =1
      \end{cases}
      \end{gather*}
  \end{questionBox}

\end{questionBox}
