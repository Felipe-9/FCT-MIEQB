                \implies &\\[6ex]&
                \implies
                \int_0^t{
                    k\,C_{A\,0}^2(1-X)^2\,V
                    \,\odif{t}
                }
                = k\,C_{A\,0}^2\,V
                \,\int_0^t{
                    \,\odif{t}
                }
                = k\,C_{A\,0}^2\,V\,t
                = &\\[3ex]&
                = \int_{1}^{1-X}{
                    N_{A\,0}
                    \frac{\odif{(1-X)}}{(1-X)^2}
                }
                = &\\&
                = N_{A\,0}
                \,\int_{1}^{1-X}{
                    \frac{\odif{(1-X)}}{(1-X)^2}
                }
                % = &\\&
                = -N_{A\,0}
                \adif{(1-X)^{-1}}
                \big\vert_{1}^{1-X}
                = &\\&
                = N_{A\,0}\left(
                    \frac{1}{1-X}
                    -1
                \right)
                = \frac{N_{A\,0}}{1/X-1}
                \implies &\\[3ex]&
                \implies
                X
                = \left(
                    1+\frac{N_{A\,0}}{k\,C_{A\,0}^2\,V\,t}
                \right)^{-1}
                % = &\\&
                = \left(
                    1+\frac{C_{A\,0}\,V}{k\,C_{A\,0}^2\,V\,t}
                \right)^{-1}
                = &\\&
                = \left(
                    1+1/k\,C_{A\,0}\,t
                \right)^{-1}
