        \frac
            {||e_1+e_2||}
            {||e_1+e_2+e_3||}

            {\sqrt{||e_1||^2+||e_2||^2}}

            {\sqrt{\sqrt{2}^2+1}}

            {\sqrt{2}}
            {\sqrt{3}}

    &
\end{flalign*}
