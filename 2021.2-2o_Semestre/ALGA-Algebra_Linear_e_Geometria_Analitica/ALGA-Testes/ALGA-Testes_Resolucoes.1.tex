\begin{questionBox}1{}

    \begin{BM}
        \{A,B,C\}\in\mathcal{M}_{n\times n}(\mathbb{R})
    \end{BM}

    \begin{questionBox}3{}
        \begin{flalign*}
            &
                A\,B = A\,C \land \det A\neq0
            \implies
                \exists A^{-1}
                \land A^{-1}\,A\,B = A^{-1}\,A\,C
            \implies
                B=C
            &
        \end{flalign*}
    \end{questionBox}

    \begin{questionBox}3{}
        \begin{flalign*}
            &
                A^2 = A\,A = I
            \implies
                \det(A)\det(A) = \det(I)
            \implies
                \det(A)^2 = 1
            \implies
                \lvert\det(A)\rvert = 1
            &
        \end{flalign*}
    \end{questionBox}

    \begin{questionBox}3{}
        \begin{flalign*}
            &
            \implies
                \det B
            =   \det(C^{-1}\,A\,C)
            =   \det(C^{-1})\,\det(A)\,\det(C)
            =   \det(A)
            &
        \end{flalign*}
    \end{questionBox}

\end{questionBox}
