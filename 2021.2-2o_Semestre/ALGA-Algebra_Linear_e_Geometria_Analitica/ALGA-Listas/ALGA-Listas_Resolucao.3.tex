                \\  1-m   & 1-m   & m^2-1
                \end{bmatrix}
            &
        \end{flalign*}
    \end{questionBox}

        \begin{flalign*}
            &
                \det M

            =   (m-1)(m(m-1)+2(m-1))
            =   (m-1)^2(m+2)
            &\\&
            \therefore
                \exists M^{-1}\forall\,M:m\in\mathbb{R}\backslash\{1,-2\}
            &
        \end{flalign*}

    \begin{BM}
        \adj(AB) = (\adj A)(\adj B)
    \end{BM}

    \begin{flalign*}
        &
            A\,B\,\adj(A\,B)=\det(A\,B)\,I_n
        \implies
            \adj(A\,B)=(A\,B)^{-1}\,\det(A\,B)\,I_n
        = &\\&
        =   \det A\,\det B\,B^{-1}\,A^{-1}\,I_n
        =   (\adj A)(\adj B)
        &
    \end{flalign*}

\end{questionBox}
