% Q3.72

\begin{BM}
    A\in\mathcal{M}_{n\times n} : A^2=-A
\end{BM}
\begin{flalign*}
    &
        \det(-A) = \det A (-1)^n = \det(A^2)
    \implies \det A (\det A - (-1)^n) = 0
    \implies \det A = 0 \lor \det A = (-1)^n
    &
\end{flalign*}

\end{BM}

\begin{flalign*}
    &
        |\det A| = 1
    =   \dots
    =   \det(A)\,\det\overline{A}^T
    =   \det(A)\,\overline{\det A^T}
    =   \det(A\,A^*) = \det (I_n) = 1
    &
\end{flalign*}

% Q3.90
\begin{BM}
    \adj(AB) = (\adj A)(\adj B)
\end{BM}

\begin{flalign*}
    &
        A\,B\,\adj(A\,B)=\det(A\,B)\,I_n
    \implies
        \adj(A\,B)=(A\,B)^{-1}\,\det(A\,B)\,I_n
    =   \det A\,\det B\,B^{-1}\,A^{-1}\,I_n
    =   (\adj A)(\adj B)
    &
\end{flalign*}
