\part{Reações de Oxidação-Redução}

\begin{sectionBox}1{Força Eletromotriz}
    
    \begin{BM}
        E = E^\circ - \frac{R\,T}{N\,\mathscr{F}}\ln(Q)
    \end{BM}

    \begin{itemize}
        \begin{multicols}{2}
            \item \(E^\circ=E^\circ_{\text{catodo}} - E^\circ_{\text{anodo}}\)
            \item \(R = 8.31446261815324\,\unit{\joule\per\kelvin\per\mole}\) Constante dos gases ideais
            \item \(\mathscr{F} = 96485.3329\,\unit{\coulomb\per\mole}\) (Exatamente) Constante de Faraday
            \item \(T\) Temperatura em kelvin
        \end{multicols}
    \end{itemize}
    
\end{sectionBox}
