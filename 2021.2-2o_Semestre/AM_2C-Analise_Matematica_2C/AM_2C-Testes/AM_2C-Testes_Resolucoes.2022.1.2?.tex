    \begin{BM}

        \left(
            \int_y^1
            x^2\,e^{x\,y}

    \end{BM}

    \begin{flalign*}
        &

                \end{aligned}
            \right\}
            \implies &\\&
            \implies
            \int_0^1

                \int_x^1

                \,\odif{y}
            \right)
            \odif{x}

            \left(
                x^2
                \left. 
                    \Delta e^{x\,y}/x 
                \right\rvert_x^1
            \right)
            \odif{x}
            = \int_0^1
            \left(
                x
                \left(
                    e^{x} - e^{x^2}
                \right)
            \right)
            \odif{x}
            = &\\&
            = \frac{e-1}{x}
        &
    \end{flalign*}
