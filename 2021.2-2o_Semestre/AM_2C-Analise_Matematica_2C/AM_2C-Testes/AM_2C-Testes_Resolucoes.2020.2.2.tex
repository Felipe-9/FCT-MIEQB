\begin{questionBox}1{}

    Descreva os pontos de estacionaridade dessa função
    \begin{BM}
        f(x,y)
        = (x-1)
        \,(y-1)
        \,(x+y)
    \end{BM}

    \vspace{-5ex}

    \begin{flalign*}
        &
            (x,y)\in A_0: 
            (0,0)
            = \nabla f(x,y)
            = &\\&
            = (
                (y-1)
                (2\,x-1+y),
                (x-1)
                (2\,y-1+x)
            )
            \implies &\\&
            \implies
            A_0 = \{
                ( 1, 1),
                (-1, 1),
                ( 1,-1)
            \}
        &
    \end{flalign*}
    
\end{questionBox}
