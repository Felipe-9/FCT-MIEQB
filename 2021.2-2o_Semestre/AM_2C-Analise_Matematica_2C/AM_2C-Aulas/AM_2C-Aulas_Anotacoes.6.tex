            \begin{aligned}

            \end{aligned}

                    + \mu\,b_2\,\hat{\jmath}
                    + \mu\,b_3\,\hat{k}
                    &
                \end{aligned}
            \right)
            = &\\&
            = \left(
                \begin{aligned}
                    & 
                    x' + \lambda\,a_1 + \mu\,b_1
                    , \\ &
                    y' + \lambda\,a_2 + \mu\,b_2
                    , \\ &
                    z' + \lambda\,a_3 + \mu\,b_3
                \end{aligned}
            \right)
        &
    \end{flalign*}
    
\end{definitionBox}

\begin{sectionBox}1{Vetor tangente a uma superficie} % 
    
    \begin{BM}[align*]
        S\subset\mathbb{R}^2 & \to\mathbb{R}^3
        \\
        (u,v) & \mapsto(\varphi,\psi,\theta)
    \end{BM}
    \vspace{-5ex}
    \begin{BM}
        T = t_1\times t_2
        = \begin{pmatrix}
            \hat{\imath} & \hat{\jmath} & \hat{k}
            \\
              \pdv{x}{u}
            & \pdv{y}{u}
            & \pdv{z}{u}
            \\
              \pdv{x}{v}
            & \pdv{y}{v}
            & \pdv{z}{v}
        \end{pmatrix}
        = 
        \left(
            \pdv{z,y}{u,v}\,\hat{\imath}
          + \pdv{z,x}{u,v}\,\hat{\jmath}
          + \pdv{x,y}{u,v}\,\hat{k}
        \right)_{(u',v')}
        \\
        T = 0 + T' + \alpha\,t_1 + \beta\,t_2 
        % = \\ 
        =
        \left(
            \begin{aligned}
                &
                x' 
                + \alpha\,\pdv{x}{u}(u',v')
                + \beta \,\pdv{x}{v}(u',v')
                , \\ &
                y' 
                + \alpha\,\pdv{y}{u}(u',v')
                + \beta \,\pdv{y}{v}(u',v')
                , \\ &
                z' 
                + \alpha\,\pdv{z}{u}(u',v')
                + \beta \,\pdv{z}{v}(u',v')
            \end{aligned}
        \right)
    \end{BM}
    
\end{sectionBox}
