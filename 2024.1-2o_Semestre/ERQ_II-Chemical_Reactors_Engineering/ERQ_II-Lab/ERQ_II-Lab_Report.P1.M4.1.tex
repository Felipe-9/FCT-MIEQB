\section{Calculos}


\begin{sectionBox}2{C2} % MARK: Calculos
    \begin{BM}
        C_{2,i+1}
        = C_{2,i}
        + \frac{C_{1,i}-C_{2,i}}{\tau}
        \frac{\beta}{(1-\alpha_1)(\alpha_2)}
        \,\adif{t}
    \end{BM}
    \begin{flalign*}
        &
            C_1\,\beta\,\nu
            = C_2\,\beta\,\nu
            + (1-\alpha_1)(\alpha_2)\,V
            \,\odv{C_2}{t}
            \implies &\\&
            \implies
            C_1\,\beta
            = C_2\,\beta
            + (1-\alpha_1)(\alpha_2)\,\tau
            \,\odv{C_2}{t}
            \implies &\\&
            \implies
            \odv{C_2}{t}
            = \frac{C_1-C_2}{\tau}
            \frac{\beta}{(1-\alpha_1)(\alpha_2)}
            \implies &\\&
            \implies
            \adv{C_2}{t}
            = \frac{C_{2,i+1}-C_{2,i}}{\adif{t}}
            = \frac{C_{1,i}-C_{2,i}}{\tau}
            \frac{\beta}{(1-\alpha_1)(\alpha_2)}
            \implies &\\&
            \implies
            = C_{2,i+1}
            = C_{2,i}
            + \frac{C_{1,i}-C_{2,i}}{\tau}
            \frac{\beta}{(1-\alpha_1)(\alpha_2)}
            \,\adif{t}
        &
    \end{flalign*}
\end{sectionBox}
