\section*{Calculos}

\begin{sectionBox}2{C2} % MARK: Calculos

    \begin{BM}
        C_{2,i+1}
        = C_{2,i}
        + \frac{C_{1,i}-C_{2,i}}{(1-\alpha_1)\,\alpha_2\,\tau}
        \,\adif{t}
    \end{BM}
    \begin{flalign*}
        &
            \nu\,C_1
            = \nu\,C_2
            + (1-\alpha_1)\,\alpha_2\,V
            \,\odv{C_2}{t}
            \implies &\\&
            \implies
            C_1
            = C_2
            + (1-\alpha_1)\,\alpha_2\,\tau
            \,\odv{C_2}{t}
            \implies &\\&
            \implies
            \odv{C_2}{t}
            = \frac{C_1-C_2}{(1-\alpha_1)\,\alpha_2\,\tau}
            \implies &\\&
            \implies
            \adv{C_2}{t}
            = \frac{C_{2,i+1}-C_{2,i}}{\adif{t}}
            = \frac{C_{1,i}-C_{2,i}}{(1-\alpha_1)\,\alpha_2\,\tau}
            \implies &\\&
            \implies
            C_{2,i+1}
            = C_{2,i}
            + \frac{C_{1,i}-C_{2,i}}{(1-\alpha_1)\,\alpha_2\,\tau}
            \,\adif{t}
        &
    \end{flalign*}
    
\end{sectionBox}

\begin{sectionBox}2{C3} % MARK: C3
    \begin{BM}
        C_{3,i+1}
        = C_{3,i}
        + \frac{C_{2,i}-C_{3,i}}{\tau\,\alpha_3\,\prod_{i=1}^{2}{(1-\alpha_i)}}
        \,\adif{t}
    \end{BM}
    \begin{flalign*}
        &
            \nu\,C_2
            = \nu\,C_3
            + \odv{C_2}{t}
            \,V\,\alpha_3\,\prod_{i=1}^{2}{(1-\alpha_i)}
            \implies &\\&
            \implies
            C_2
            = C_3
            + \odv{C_3}{t}
            \,\tau\,\alpha_3\,\prod_{i=1}^{2}{(1-\alpha_i)}
            \implies &\\&
            \implies
            \odv{C_3}{t}
            = \frac{C_2-C_3}{\tau\,\alpha_3\,\prod_{i=1}^{2}{(1-\alpha_i)}}
            \implies &\\&
            \implies
            \adv{C_3}{t}
            = \frac{C_{3,i+1}-C_{3,i}}{\adif{t}}
            = \frac{C_{2,i}-C_{3,i}}{\tau\,\alpha_3\,\prod_{i=1}^{2}{(1-\alpha_i)}}
            \implies &\\&
            \implies
            C_{3,i+1}
            = C_{3,i}
            + \frac{C_{2,i}-C_{3,i}}{\tau\,\alpha_3\,\prod_{i=1}^{2}{(1-\alpha_i)}}
            \,\adif{t}
        &
    \end{flalign*}
\end{sectionBox}
