\part*{C1}
\begin{minipage}{140mm}
    \begin{BM}
        C_1
        =E(t)
        = \frac{\sqrt{Pe}}{
            2\,\sqrt{
                \pi\,(t/\alpha\,\tau)^3
            }
        }\,\exp{\left(
            -\frac{Pe\,\alpha\,\tau}{4\,t}
            \left(
                \frac{t}{\alpha\,\tau}-1
            \right)^2
        \right)}
    \end{BM}

\part*{C out}
\begin{minipage}{1mm}
    \begin{BM}
        C_{out,i+1}
        = C_{out,i}
        + \frac{C_{1,i}-C_{out,i}}{(1-\alpha)\,\tau}
        \,\adif{t}
    \end{BM}
    \eqsep
    \begin{flalign*}
        &
            \nu\,C_1
            =\nu\,C_{out}+(1-\alpha)\,V\,\odv{C_{out}}{t}
            \implies &\\&
            \implies
            C_1
            =C_{out}+(1-\alpha)\,\tau\,\odv{C_{out}}{t}
            \implies &\\&
            \implies
            \adv{C_{out}}{t}
            = \frac{C_{out,i+1}-C_{out,i}}{\adif{t}}
            = \frac{C_1-C_{out}}{(1-\alpha)\,\tau}
            \implies &\\&
            \implies
            C_{out,i+1}
            = C_{out,i}
            + \frac{C_1-C_{out}}{(1-\alpha)\,\tau}
            \,\adif{t}
        &
    \end{flalign*}
\end{minipage}
