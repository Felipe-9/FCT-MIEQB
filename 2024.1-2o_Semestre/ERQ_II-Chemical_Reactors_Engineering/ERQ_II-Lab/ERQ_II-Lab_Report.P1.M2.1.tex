\section*{Calculos}

\begin{sectionBox}2{C2} % MARK: Calculos
    
    \begin{BM}
        C_{2,i+1}
        = C_{2,i}
        + \frac{C_{1,i}-C_{2,i}}{\tau}
        \,\frac{\beta}{(1-\alpha_1)\,(1-\alpha_1)\,\alpha_2}
        \,\adif{t}
    \end{BM}
    \begin{flalign*}
        &
            \beta\,\nu\,C_{1}
            = \beta\,\nu\,C_{2}
            + (1-\alpha_1)\,\alpha_2\,V\,\odv{C_{2}}{t}
            \implies &\\&
            \implies
            \beta\,C_{1}
            = \beta\,C_{2}
            + (1-\alpha_1)\,\alpha_2\,\tau\,\odv{C_{2}}{t}
            \implies &\\&
            \implies
            \odv{C_{2}}{t}
            = \frac{C_{1}-C_{2}}{(1-\alpha_1)\,\alpha_2\,\tau}
            \,\beta
            \implies &\\&
            \implies
            \adv{C_{2}}{t}
            = \frac{C_{2,i+1}-C_{2,i}}{\adif{t}}
            = \frac{C_{1,i}-C_{2,i}}{(1-\alpha_1)\,\alpha_2\,\tau}
            \,\beta
            \implies &\\&
            \implies
            C_{2,i+1}
            = C_{2,i}
            + \frac{C_{1,i}-C_{2,i}}{(1-\alpha_1)\,\alpha_2\,\tau}
            \,\beta\,\adif{t}
        &
    \end{flalign*}
    
\end{sectionBox}

\begin{sectionBox}2{C3} % MARK: S
    
    \begin{BM}
        C_3 = \beta\,C_2 + (1-\beta)\,C_1
    \end{BM}
    \begin{flalign*}
        &
            \nu\,C_3
            = \beta\,\nu\,C_2
            + (1-\beta)\,\nu\,C_1
            \implies &\\&
            \implies
            C_3 = \beta\,C_2 + (1-\beta)\,C_1
        &
    \end{flalign*}
    
\end{sectionBox}

\begin{sectionBox}2{C out} % MARK: S
    
    \begin{BM}
        C_{out,i+1}
        = C_{out,i}
        + \frac{C_{out,i}-C_{3,i}}{(1-\alpha_1)(1-\alpha_2)\,\tau}
        \,\adif{t}
    \end{BM}
    \begin{flalign*}
        &
            \nu\,C_{out}
            = \nu\,C_{3}
            + (1-\alpha_1)(1-\alpha_2)\,V
            \,\odv{C_{out}}{t}
            \implies &\\&
            \implies
            C_{out}
            = C_{3}
            + (1-\alpha_1)(1-\alpha_2)\,\tau
            \,\odv{C_{out}}{t}
            \implies &\\&
            \implies
            \odv{C_{out}}{t}
            = \frac{C_{out}-C_{3}}{(1-\alpha_1)(1-\alpha_2)\,\tau}
            \implies &\\&
            \implies
            \adv{C_{out}}{t}
            = \frac{C_{out,i+1}-C_{out,i}}{\adif{t}}
            = \frac{C_{out,i}-C_{3,i}}{(1-\alpha_1)(1-\alpha_2)\,\tau}
            \implies &\\&
            \implies
            C_{out,i+1}
            = C_{out,i}
            + \frac{C_{out,i}-C_{3,i}}{(1-\alpha_1)(1-\alpha_2)\,\tau}
            \,\adif{t}
        &
    \end{flalign*}
    
\end{sectionBox}
